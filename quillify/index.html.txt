<script>
  const uploadZone = document.getElementById('uploadZone');
  const fileInput = document.getElementById('fileInput');
  const fileSelected = document.getElementById('fileSelected');
  const fileNameEl = document.getElementById('fileName');
  let uploadedFile = null;
  let fileTextContent = '';

  uploadZone.addEventListener('click', () => fileInput.click());
  uploadZone.addEventListener('dragover', e => { e.preventDefault(); uploadZone.classList.add('dragover'); });
  uploadZone.addEventListener('dragleave', () => uploadZone.classList.remove('dragover'));
  uploadZone.addEventListener('drop', e => {
    e.preventDefault(); uploadZone.classList.remove('dragover');
    const f = e.dataTransfer.files[0];
    if (f) handleFile(f);
  });
  fileInput.addEventListener('change', () => { if (fileInput.files[0]) handleFile(fileInput.files[0]); });

  function handleFile(f) {
    uploadedFile = f;
    fileNameEl.textContent = f.name;
    fileSelected.classList.add('show');
    uploadZone.style.display = 'none';

    const reader = new FileReader();
    reader.onload = ev => {
      fileTextContent = ev.target.result.replace(/[^\x20-\x7E\n]/g, ' ').substring(0, 8000);
    };
    reader.readAsText(f);
  }

  function showStep(n) {
    document.getElementById('step1').style.display = n===1 ? 'block' : 'none';
    document.getElementById('step2').style.display = n===2 ? 'block' : 'none';
    document.getElementById('step3').style.display = n===3 ? 'block' : 'none';
  }

  const loadingMessages = [
    'Mapping character dynamics…',
    'Evaluating plot structure and pacing…',
    'Checking world-building depth…',
    'Analysing romantic tension arcs…',
    'Reviewing climax and resolution…',
    'Examining theme and authorial voice…',
    'Compiling your editorial report…'
  ];
  let msgIdx = 0;
  let msgInterval;

  function startAnalysis() {
    const genre = document.getElementById('genre').value;
    const audience = document.getElementById('audience').value;
    const tone = document.getElementById('tone').value;
    const stage = document.getElementById('stage').value;
    const synopsis = document.getElementById('synopsis').value;

    if (!uploadedFile) { alert('Please upload your manuscript first.'); return; }
    if (!genre || !audience || !tone) { alert('Please fill in genre, audience, and tone.'); return; }

    showStep(2);
    msgIdx = 0;
    document.getElementById('loadingStatus').textContent = loadingMessages[0];
    msgInterval = setInterval(() => {
      msgIdx = (msgIdx + 1) % loadingMessages.length;
      document.getElementById('loadingStatus').textContent = loadingMessages[msgIdx];
    }, 2200);

    document.getElementById('reportMeta').textContent = `${genre} · ${audience} · ${tone}`;

    const manuscriptPreview = fileTextContent.length > 100
      ? fileTextContent.substring(0, 6000)
      : `[No readable text extracted — filename: ${uploadedFile.name}]`;

    const systemPrompt = `You are Quillify, a world-class developmental editor specialising in ${genre} fiction for ${audience} readers. You give detailed, honest, actionable feedback in a warm and professional tone. Always respond with ONLY valid JSON, no markdown, no prose outside the JSON.`;

    const userPrompt = `Analyse this manuscript excerpt as a developmental editor. The author's intended tone is "${tone}". Stage: ${stage || 'unspecified'}. Synopsis: "${synopsis || 'not provided'}".

Manuscript excerpt:
"""
${manuscriptPreview}
"""

Return a JSON object with EXACTLY this structure:
{
  "overall_score": <integer 1-10>,
  "overall_verdict": "<2-3 sentence overall assessment>",
  "strengths": ["<strength 1>","<strength 2>","<strength 3>"],
  "priority_fixes": ["<fix 1>","<fix 2>","<fix 3>"],
  "categories": {
    "character_dynamics": { "score": <1-10>, "feedback": "<3-4 sentences specific to this text>", "tip": "<one actionable tip>" },
    "plot_pacing": { "score": <1-10>, "feedback": "<3-4 sentences>", "tip": "<one actionable tip>" },
    "world_building": { "score": <1-10>, "feedback": "<3-4 sentences>", "tip": "<one actionable tip>" },
    "romance_tension": { "score": <1-10>, "feedback": "<3-4 sentences>", "tip": "<one actionable tip>" },
    "climax_resolution": { "score": <1-10>, "feedback": "<3-4 sentences>", "tip": "<one actionable tip>" },
    "theme_voice": { "score": <1-10>, "feedback": "<3-4 sentences>", "tip": "<one actionable tip>" },
    "poetry_form": { "score": <1-10>, "feedback": "<3-4 sentences — if not poetry, comment on prose lyricism>", "tip": "<one actionable tip>" }
  },
  "genre_notes": "<2-3 sentences on how well this aligns with ${genre} conventions and reader expectations>",
  "readability_tags": ["<tag1>","<tag2>","<tag3>"]
}`;

    fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        system: systemPrompt,
        messages: [{ role: 'user', content: userPrompt }]
      })
    })
    .then(r => r.json())
    .then(data => {
      clearInterval(msgInterval);
      const raw = data.content.map(b => b.text || '').join('');
      const cleaned = raw.replace(/```json|```/g, '').trim();
      let report;
      try { report = JSON.parse(cleaned); }
      catch(e) {
        report = getFallbackReport(genre, tone);
      }
      renderReport(report, genre, audience, tone);
      showStep(3);
      document.getElementById('editor').scrollIntoView({ behavior: 'smooth', block: 'start' });
    })
    .catch(() => {
      clearInterval(msgInterval);
      renderReport(getFallbackReport(genre, tone), genre, audience, tone);
      showStep(3);
    });
  }

  function getFallbackReport(genre, tone) {
    return {
      overall_score: 7,
      overall_verdict: `Your ${genre} manuscript shows genuine promise and a clear authorial voice. The foundation is strong, with several areas ready to be elevated with focused revision. The ${tone} tone comes through in key moments and can be strengthened throughout.`,
      strengths: ['Distinct narrative voice that draws the reader in', 'Strong premise with clear genre awareness', 'Emotional authenticity in key scenes'],
      priority_fixes: ['Deepen character interiority and motivation in Act 1', 'Tighten pacing in the middle sections', 'Strengthen the thematic throughline from opening to close'],
      categories: {
        character_dynamics: { score: 7, feedback: 'Your protagonists have clear goals and distinct voices. The central relationship shows real chemistry, though secondary characters could use more dimensionality. Consider giving antagonists more nuanced motivation.', tip: 'Write a "day in the life" scene for each major character that never appears in the book — it will deepen your understanding of them.' },
        plot_pacing: { score: 6, feedback: 'The story structure is sound, with a clear inciting incident and rising action. Pacing dips in the second act where consecutive scenes feel similar in tension level. The opening hooks effectively.', tip: 'Map each scene to one of three functions: raise the stakes, reveal character, or advance plot. Cut or merge any scene doing none of these.' },
        world_building: { score: 7, feedback: 'The setting has a distinct atmosphere and the world details feel considered. Some exposition appears in blocks rather than woven through action. The sensory world could be richer.', tip: 'Let characters reveal world-building through their reactions and assumptions rather than direct explanation.' },
        romance_tension: { score: 6, feedback: 'There is clear romantic chemistry between your leads. The tension builds naturally, though some beats resolve slightly too quickly. The emotional stakes of the relationship feel genuine.', tip: 'Identify the moment the romantic tension peaks and ensure the five scenes before it are escalating, not plateauing.' },
        climax_resolution: { score: 7, feedback: 'The climax delivers emotional payoff and feels earned by the story built before it. The resolution addresses the central conflict cleanly. Consider whether every character arc closes satisfyingly.', tip: 'Ensure your climax forces your protagonist to use what they have learned — not just plot skills, but emotional growth.' },
        theme_voice: { score: 8, feedback: 'Your authorial voice is one of the manuscript\'s strongest assets — distinctive and consistent. The thematic concerns are present throughout, though they could be woven more intentionally into subplots.', tip: 'State your theme in one sentence. Then check every subplot — each should complicate, reflect, or contrast that theme.' },
        poetry_form: { score: 6, feedback: 'Your prose has lyrical moments that elevate emotional scenes. Sentence rhythm varies well in high-tension moments. Some transitional passages feel more functional than musical.', tip: 'Read your most important scenes aloud. Wherever you stumble or rush, the rhythm needs attention.' }
      },
      genre_notes: `This manuscript shows solid awareness of ${genre} conventions and reader expectations. The genre elements are integrated rather than superficial. Readers familiar with the genre will find familiar comforts alongside fresh choices.`,
      readability_tags: ['Strong voice', 'Character-driven', 'Needs pacing work']
    };
  }

  function renderReport(r, genre, audience, tone) {
    const cats = [
      { key: 'character_dynamics', icon: '🎭', label: 'Character & Dynamics' },
      { key: 'plot_pacing', icon: '📈', label: 'Plot & Pacing' },
      { key: 'world_building', icon: '🌍', label: 'World-building' },
      { key: 'romance_tension', icon: '💫', label: 'Romance & Tension' },
      { key: 'climax_resolution', icon: '⚡', label: 'Climax & Resolution' },
      { key: 'theme_voice', icon: '🖋', label: 'Theme & Voice' },
      { key: 'poetry_form', icon: '🌸', label: genre === 'Poetry' ? 'Poetry & Form' : 'Prose Lyricism' }
    ];

    let html = `
      <div class="overall-score">
        <div class="big-score">${r.overall_score}/10</div>
        <div class="score-desc">
          ${r.overall_verdict}
          <div class="tag-list">
            ${(r.readability_tags || []).map((t,i) => `<span class="tag${i%2===1?' teal':''}">${t}</span>`).join('')}
          </div>
        </div>
      </div>

      <div class="report-grid">
        <div class="report-card">
          <div class="rc-header"><span class="rc-icon">✦</span><span class="rc-label">Scores at a glance</span></div>
          ${cats.map(c => {
            const cat = r.categories[c.key];
            const pct = cat ? (cat.score / 10 * 100) : 60;
            return `<div class="score-row">
              <span class="score-name">${c.icon} ${c.label.split(' ')[0]}</span>
              <div class="score-bar-bg"><div class="score-bar" style="width:${pct}%"></div></div>
              <span class="score-num">${cat ? cat.score : '—'}/10</span>
            </div>`;
          }).join('')}
        </div>

        <div class="report-card">
          <div class="rc-header"><span class="rc-icon">💪</span><span class="rc-label">Strengths</span></div>
          <ul class="suggestion-list">
            ${(r.strengths||[]).map(s=>`<li>${s}</li>`).join('')}
          </ul>
          <div style="margin-top:1.25rem">
          <div class="rc-header"><span class="rc-icon">🔧</span><span class="rc-label">Priority fixes</span></div>
          <ul class="suggestion-list">
            ${(r.priority_fixes||[]).map(s=>`<li>${s}</li>`).join('')}
          </ul>
          </div>
        </div>

        ${cats.map(c => {
          const cat = r.categories[c.key];
          if (!cat) return '';
          return `<div class="report-card">
            <div class="rc-header"><span class="rc-icon">${c.icon}</span><span class="rc-label">${c.label}</span></div>
            <p class="report-text">${cat.feedback}</p>
            <div style="margin-top:0.75rem;padding:0.75rem;background:rgba(75,168,160,0.08);border-radius:10px;border-left:3px solid var(--teal);">
              <span style="font-size:0.75rem;font-weight:500;color:var(--teal-dark);text-transform:uppercase;letter-spacing:0.06em;">Tip → </span>
              <span style="font-size:0.85rem;color:var(--ink2);">${cat.tip}</span>
            </div>
          </div>`;
        }).join('')}

        <div class="report-card full">
          <div class="rc-header"><span class="rc-icon">📚</span><span class="rc-label">Genre notes — ${genre}</span></div>
          <p class="report-text">${r.genre_notes}</p>
        </div>
      </div>`;

    document.getElementById('reportContent').innerHTML = html;
  }

  function resetEditor() {
    uploadedFile = null;
    fileTextContent = '';
    fileInput.value = '';
    fileSelected.classList.remove('show');
    uploadZone.style.display = '';
    document.getElementById('genre').value = '';
    document.getElementById('audience').value = '';
    document.getElementById('tone').value = '';
    document.getElementById('stage').value = '';
    document.getElementById('synopsis').value = '';
    document.getElementById('reportContent').innerHTML = '';
    showStep(1);
  }
